# Célula 2: Importação e Limpeza "À Prova de Falhas"
# Explicação: Nesta etapa, convertemos o arquivo Excel em DataFrames e removemos acentos e espaços dos nomes das colunas para evitar erros de digitação.

url = "https://github.com/Montcharles/aula_descritiva/raw/main/Dados%20Ecol%C3%B3gicos%20Descritiva.xlsx"

def robust_clean(df):
    # Remove acentos, espaços e coloca tudo em minúsculas
    df.columns = (df.columns.str.normalize('NFKD')
                  .str.encode('ascii', errors='ignore')
                  .str.decode('utf-8')
                  .str.replace(' ', '_')
                  .str.replace('(', '')
                  .str.replace(')', '')
                  .str.lower())
    return df

# Baixando e limpando cada aba
response = requests.get(url)
file = io.BytesIO(response.content)

morpho = robust_clean(pd.read_excel(file, sheet_name=0))
riqueza = robust_clean(pd.read_excel(file, sheet_name=1))
ambiental = robust_clean(pd.read_excel(file, sheet_name=2))
cresc_pop = robust_clean(pd.read_excel(file, sheet_name=3))

print("✅ Colunas detectadas em Morpho:", morpho.columns.tolist())

#Célula 7: Violin Plot (Densidade + Estatística)
# O Violin Plot é uma evolução do Boxplot. Ele mostra a "forma" da distribuição (onde há mais indivíduos) através da largura do desenho. É ideal para identificar se a distribuição é multimodal (dois picos).

plt.figure(figsize=(12, 6))

# Comparando a distribuição da massa por espécie e sexo
sns.violinplot(data=morpho, x='especie', y='massa_g', hue='sexo', 
               split=True, inner="quart", palette="Pastel1")

plt.title('Distribuição de Densidade da Massa (Violin Plot)')
plt.ylabel('Massa (g)')
sns.despine()
plt.show()





#Célula 8: Gráfico de Barras com Intervalo de Confiança
# Diferente do boxplot, o gráfico de barras foca na estimativa da média. O "risquinho" no topo (error bar) representa o intervalo de confiança ou desvio padrão, mostrando o quanto podemos confiar nessa média.


plt.figure(figsize=(10, 6))

# Massa média por habitat com barra de erro (CI 95%)
sns.barplot(data=morpho, x='habitat', y='massa_g', hue='especie', 
            capsize=.1, palette="viridis")

plt.title('Massa Média por Habitat (Com Intervalo de Confiança)')
plt.ylabel('Massa Média (g)')
plt.legend(title='Espécie', bbox_to_anchor=(1, 1))
plt.show()




# Célula 9: Point Plot (Análise de Interação)
# Este gráfico é excelente para ver tendências. Ele conecta as médias com linhas, facilitando a visualização de como o Sexo e o Habitat interagem. Se as linhas se cruzam, há uma forte indicação de que o ambiente afeta machos e fêmeas de forma diferente.

plt.figure(figsize=(10, 6))

# Analisando se a diferença de massa entre sexos muda conforme o habitat
sns.pointplot(data=morpho, x='habitat', y='massa_g', hue='sexo', 
              markers=["o", "x"], linestyles=["-", "--"])

plt.title('Tendência de Massa: Interação Sexo vs Habitat')
plt.ylabel('Massa (g)')
plt.grid(axis='y', linestyle='--', alpha=0.7)
plt.show()





# Célula 10: Pair Plot (Matriz de Dispersão Completa)
# Para encerrar a exploração morfométrica, usamos o "gráfico de tudo contra tudo". Ele gera automaticamente histogramas na diagonal e gráficos de dispersão para todos os pares de variáveis numéricas. É a ferramenta definitiva para encontrar padrões inesperados.


# Analisando Comprimento, Massa e Envergadura de uma só vez
# 'kind=reg' adiciona uma reta de regressão em cada gráfico
g = sns.pairplot(morpho, vars=['comprimento_cm', 'massa_g', 'envergadura_cm'], 
                 hue='especie', palette='husl', diag_kind='kde', height=3)

g.fig.suptitle('Matriz de Dispersão Geral: Relações Morfométricas', y=1.02)
plt.show()

print("🎯 Exploração Morfométrica Finalizada.")
